      cmatcher = generalEntity(nlp, patterns)
      doc = nlp("Vibration seems like it is coming from the shaft.")

      nlp.add_pipe('general_entity', config={"patterns": {'label': 'general', 'pattern': [{'LOWER': 'possible'}], 'id': 'general'}, "asSpan":True})
